public static int duplicationNum2(int[] nums) {
        Arrays.sort(nums);
        int slow = 0, fast = 1;
        do {
            if (nums[slow] == nums[fast]) {
                return nums[slow];
            }
            ++slow;
            ++fast;
        } while (fast <= nums.length);
        return nums.length;
